/* 开头一个块注释，包含各种符号和单词，测试块注释跳过 */
int/* and some tricky symbols : ( ) { } ; */ factorial (int n) {
    int result = 1; /* with multi-line comment
                        that contains stars * and slashes / and words */
    // 一个行注释，包含运算符和关键字 if x <= 0 { return 1; }
    if (n <= 0) { 
        return 1; 
    } else {
        /* 注释里包含类似注释起始符号的文本 "/* not real start" */
        while (n > 1) {
            result = result * n; /* multiply */
            n = n - 1; // decrement n
        }
        return result;
    }
}
